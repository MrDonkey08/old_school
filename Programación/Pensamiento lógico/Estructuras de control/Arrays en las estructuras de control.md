# Arreglos en las estructuras de control

## Con ciclo for

	edades = [a, b, c, d]
	for (i = 0; i < 4; i++)
		if edades[i] >=18
			"Es mayor de edad."
		else
			"Es menor de edad."


## Con ciclo while

	edades = [a, b, c, d]
	cont = 0
	while cont <= 3
	cont = cont + 1
		if edades[i] >=18
			"Es mayor de edad."
		else
			"Es menor de edad."

## Con ciclo dowhile
	edades = [a, b, c, d]
	cont = 0
	do
		cont = cont + 1
		if edades[i] >=18
			"Es mayor de edad."
		else
			"Es menor de edad."
	while cont <= 3
